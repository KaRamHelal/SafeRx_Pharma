## OpenAI Function Calling

```json
{
  "type": "function",
  "function": {
    "name": "check_drug_safety",
    "description": "Screen drugs for safety issues including adverse effects, drug interactions, pregnancy/lactation risks, food interactions, clinical considerations, and dosing. Covers 28,000+ Egyptian pharmaceutical products.",
    "strict": true,
    "parameters": {
      "type": "object",
      "required": ["drugs"],
      "additionalProperties": false,
      "properties": {
        "drugs": {
          "type": "array",
          "items": {"type": "string"},
          "description": "Drug names (trade or generic, max 20)"
        },
        "patient_profile": {
          "type": "object",
          "properties": {
            "populations": {
              "type": "array",
              "items": {"type": "string", "enum": ["pediatric", "geriatric", "cardiac", "hepatic", "renal"]}
            },
            "conditions": {
              "type": "array",
              "items": {"type": "string", "enum": ["diabetes", "hypertension", "epilepsy", "heart_failure", "asthma", "liver_disease", "kidney_disease", "thyroid", "depression", "blood_disorders", "ischemic_heart", "arrhythmia", "gout", "osteoporosis"]}
            }
          },
          "additionalProperties": false
        },
        "include": {
          "type": "array",
          "items": {"type": "string", "enum": ["ae", "ddi", "pllr", "food", "clinical", "dose"]},
          "description": "Safety domains to check (omit for all)"
        },
        "lang": {
          "type": "string",
          "enum": ["en", "ar"],
          "description": "Response language"
        }
      }
    }
  }
}
```
